// 1. 在全局定义一个 name
var name = "我是Window"; 

const obj = {
    name: "我是Object",

    // 情况 A：普通函数
    sayNormal: function() {
        console.log(this.name);
    },

    // 情况 B：箭头函数
    sayArrow: () => {
        console.log(this.name);
    }
};

// --- 请预测输出 ---
obj.sayNormal(); // "我是Object",
obj.sayArrow();  //  前面var name = "我是Window"; 的话就是"我是Window"； 用const就是undefined；
