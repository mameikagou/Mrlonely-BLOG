
最长递增子序列 (LIS) - LC 300
/**
 * @param {number[]} nums
 * @return {number}
 */
var lengthOfLIS = function(nums) {
    if(!nums) return 0;
    // 基于背包的解法
    const len = nums.length;
    const dp = new Array(len+1).fill(1);

    let maxLen = 1;
    for(let i=1;i<len;i++){
        for(let j=0;j<i;j++){
            if(nums[i]>nums[j]){
                dp[i] = Math.max(dp[i], dp[j]+1)
            }
        }
        maxLen = Math.max(maxLen, dp[i]);
    }

    return maxLen;
};
