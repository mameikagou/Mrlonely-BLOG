
##  /bin, /boot, /ect


看一个计算一个目录下所有内容的例子

```sh
#!/bin/bash

calculate_total_size() {
    local directory="$1"

    # 这里表if，要有 if [];then fi 
    # d表示dictionary
    if [ ! -d "$directory" ]; then
        echo "$directory is not a directory"
        exit 1
    fi

    total_size=$(find "$directory" -type f -exec du -b {} \; | awk '{sum += $1} END {print sum}')
    echo "Total size of $directory: $total_size bytes"


}

echo "请输入目录路径: "
read directory
calculate_total_size "$directory"
```
